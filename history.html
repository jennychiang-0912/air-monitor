const connectBtn = document.getElementById("connectBtn"); const statusText = document.getElementById("status"); const output = document.getElementById("output"); const queryBtn = document.getElementById("queryBtn"); const exportBtn = document.getElementById("exportBtn"); const clearBtn = document.getElementById("clearBtn"); let port; let reader; connectBtn.onclick = async () => { port = await navigator.serial.requestPort(); await port.open({ baudRate: 9600 }); statusText.textContent = "已連接 Arduino"; readSerial(); }; async function sendCommand(cmd) { const encoder = new TextEncoder(); const writer = port.writable.getWriter(); await writer.write(encoder.encode(cmd + "\n")); writer.releaseLock(); } let csvBuffer = ""; let isCSV = false; async function readSerial() { const decoder = new TextDecoderStream(); port.readable.pipeTo(decoder.writable); reader = decoder.readable.getReader(); while (true) { const { value } = await reader.read(); const lines = value.split("\n"); for (let line of lines) { line = line.trim(); if (!line) continue; if (line === "CSV_BEGIN") { isCSV = true; csvBuffer = ""; continue; } if (line === "CSV_END") { isCSV = false; output.textContent = csvBuffer; downloadCSV(csvBuffer); continue; } if (isCSV) { csvBuffer += line + "\n"; } } } } queryBtn.onclick = async () => { const start = document.getElementById("startTime").value; const end = document.getElementById("endTime").value; if (!start || !end) { alert("請輸入時間"); return; } output.textContent = "查詢中..."; await sendCommand(`QUERY,${start},${end}`); }; exportBtn.onclick = async () => { output.textContent = "匯出中..."; await sendCommand("EXPORT"); }; clearBtn.onclick = async () => { if (!confirm("確定清除資料？")) return; await sendCommand("CLEAR"); output.textContent = "已清除"; }; function downloadCSV(text) { const blob = new Blob([text], { type: "text/csv" }); const url = URL.createObjectURL(blob); const a = document.createElement("a"); a.href = url; a.download = "air_data.csv"; a.click(); URL.revokeObjectURL(url); }
